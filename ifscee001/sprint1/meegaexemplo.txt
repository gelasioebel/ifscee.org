#include <stdio.h>

// 1. TESTE: Função Recursiva (Testa a Pilha de Chamadas e o 'if/else')
int fatorial(int n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * fatorial(n - 1); // Chamada recursiva!
    }
}

// 2. TESTE: Manipulação por Referência (Ponteiros como parâmetros)
int dobrar_valor(int* ptr) {
    *ptr = *ptr * 2; // Desreferenciação e modificação
    return *ptr;
}

int main() {
    // 3. TESTE: Strings Literais (Área RODATA)
    char* titulo = "=== TESTE DE STRESS IFSCee ===";
    printf(titulo);
    printf("\n");

    // 4. TESTE: Recursão na Prática
    int fat = fatorial(4); // Vai empilhar até n=1 e retornar 24
    printf("Fatorial de 4 (esperado 24): ");
    printf("%d", fat);
    printf("\n");

    // 5. TESTE: For Loop, Arrays e Atribuição Composta (+=)
    int numeros[3];
    int soma = 0;
    for (int i = 0; i < 3; i++) {
        numeros[i] = i + 5; // Preenche com 5, 6 e 7
        soma += numeros[i]; // Somatório: 5+6+7 = 18
    }
    printf("Soma do Array (esperado 18): ");
    printf("%d", soma);
    printf("\n");

    // 6. TESTE: Operador Endereço de (&)
    int alvo = 10;
    int resultado_dobro = dobrar_valor(&alvo);
    printf("Dobro por referencia (esperado 20): ");
    printf("%d", alvo); // Imprime 20, provando que a função alterou a Stack do main!
    printf("\n");

    // 7. TESTE: While, Continue, Break e Lógica (|| e &&)
    int contador = 0;
    while (contador < 10) {
        contador++; // Operador sufixo testado!

        // Curto-circuito OR
        if (contador == 2 || contador == 3) {
            continue; // Ignora o resto e volta para o topo do while
        }

        // Curto-circuito AND
        if (contador == 5 && alvo == 20) {
            break; // Estoura o laço imediatamente
        }
    }
    printf("Contador parou no Break (esperado 5): ");
    printf("%d", contador);
    printf("\n");

    // 8. TESTE: Do-While e Decremento (--)
    int dw = 3;
    do {
        dw--;
    } while (dw > 0);
    printf("Do-While final (esperado 0): ");
    printf("%d", dw);
    printf("\n");

    // 9. TESTE: Malloc, Heap e Operador Lógico (!)
    int* dinamico = malloc(8); // Aloca 8 bytes (2 ints)
    dinamico[0] = 77; // Acesso a Heap via notação de array
    dinamico[1] = 88;

    if (!0) { // !0 é o mesmo que VERDADEIRO em C
        printf("Primeiro bloco do Heap (esperado 77): ");
        printf("%d", *dinamico); // *dinamico é o mesmo que dinamico[0]
        printf("\n");
    }

    free(dinamico); // Libera o bloco do Heap

    printf("=== FIM DOS TESTES ===\n");
    return 0; // Finaliza com Exit Code 0
}
