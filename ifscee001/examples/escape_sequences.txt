// =============================================================================
// EXAMPLE 1: Basic Escape Sequences
// Demonstrates: \n (newline), \t (tab), \\ (backslash), \" (quote)
// =============================================================================
#include <stdio.h>

int main() {
    printf("Line 1\nLine 2\nLine 3");
    printf("Tabbed:\tValue1\tValue2");
    printf("Backslash: \\");
    printf("Quote: \"Hello\"");
    return 0;
}
